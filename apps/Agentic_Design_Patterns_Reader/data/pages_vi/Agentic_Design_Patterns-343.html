class ProfessorAgent(BaseAgent):
def __init__(self, model="gpt4omini", notes=None, max_steps=100,
openai_api_key=None):
super().__init__(model, notes, max_steps, openai_api_key)
self.phases = ["report writing"]
def generate_readme(self):
sys_prompt = f"""Bạn là {self.role_description()} \n Đây là
bài báo đã viết \n{self.report}. Hướng dẫn nhiệm vụ: Mục tiêu của bạn là
tích hợp tất cả kiến thức, code, báo cáo và ghi chú được cung cấp cho
bạn và tạo readme.md cho kho lưu trữ github."""
history_str = "\n".join([_[1] for _ in self.history])
prompt = (
f"""Lịch sử: {history_str}\n{'~' * 10}\n"""
f"Vui lòng tạo readme bên dưới theo markdown:\n")
model_resp = query_model(model_str=self.model,
system_prompt=sys_prompt, prompt=prompt,
openai_api_key=self.openai_api_key)
return model_resp.replace("```markdown", "")
