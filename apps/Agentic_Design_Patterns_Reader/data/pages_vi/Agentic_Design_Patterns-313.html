Tham số:
survey_question (str): Câu hỏi khảo sát pháp lý cần được đánh giá.
Trả về:
Optional[dict]: Một từ điển chứa phán đoán của LLM,
hoặc None nếu có lỗi xảy ra.
"""

try:
logging.info(f"Đang gửi yêu cầu tới
'{self.model.model_name}' để phán xét...")

full_prompt,
generation_config=genai.types.GenerationConfig(
temperature=self.temperature,
response_mime_type="application/json"
)
)
# Kiểm tra kiểm duyệt nội dung hoặc các lý do khác cho phản hồi
trống.
if not response.parts:
safety_ratings =
response.prompt_feedback.safety_ratings
logging.error(f"Phản hồi LLM trống hoặc bị chặn.
Đánh Giá An Toàn: {safety_ratings}")
return None
return json.loads(response.text)
except json.JSONDecodeError:
logging.error(f"Không thể giải mã phản hồi LLM thành JSON. Phản hồi thô:
{response.text}")
return None
except Exception as e:
logging.error(f"Xảy ra lỗi không mong đợi trong quá trình phán xét LLM:
{e}")
return None
